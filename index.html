// Google Apps Script — 매일 일본어 미스터리 대본 자동 생성
const CLAUDE_KEY = 'sk-ant-여기에_키_입력';

function generateDailyScripts() {
  const sheet = SpreadsheetApp.getActiveSheet();
  const prompt = '일본어 미스터리 숏츠 대본 3개를 JSON 배열로. ' +
    '각 항목: hook(훅3줄), body(본문), korean(한국어번역), invideo_prompt(영어 영상프롬프트). ' +
    'JSON만 출력.';

  const res = UrlFetchApp.fetch('https://api.anthropic.com/v1/messages', {
    method: 'post',
    headers: { 'x-api-key': CLAUDE_KEY, 'anthropic-version': '2023-06-01' },
    contentType: 'application/json',
    payload: JSON.stringify({
      model: 'claude-sonnet-4-6',
      max_tokens: 4000,
      messages: [{ role: 'user', content: prompt }]
    })
  });

  const data = JSON.parse(res.getContentText());
  let text = data.content[0].text.replace(/```json|```/g, '').trim();
  const scripts = JSON.parse(text.slice(text.indexOf('['), text.lastIndexOf(']') + 1));

  scripts.forEach(s => {
    sheet.appendRow([
      new Date(), s.hook, s.body, s.korean, s.invideo_prompt, '미업로드'
    ]);
  });
}
